### 3. Embedding Surface Form

<Tabs>
  <Tab title="Inline">
    1. In framer, click on `insert` to add new component inside the website.
    2. Select `Embed` from the list of components.
        <Accordion title="Click to expand">
          <img className="block" src="/images/framer/4.png" alt="Hero Light" />
        </Accordion>
    3. Insert the component in the desired location. Now go to `properties` of the component and scroll down to `Embed` section.
    4. Set the `Type` to `URL` and paste you're form URL in the `URL` field.
        <Accordion title="Click to expand">
          <img className="block" src="/images/framer/3.png" alt="Hero Light" />
        </Accordion>
    5. That should look something like this:
        <Accordion title="Click to expand">
          <img className="block" src="/images/framer/2.png" alt="Hero Light" />
        </Accordion> 
  </Tab>
  <Tab title="Popover or Slideover">
    <Tip>This feature requires paid tier of framer!</Tip>
    1. In framer, Go to `settings` of the project.
    2. Scroll down to `Custom Code` section.
        <Accordion title="Click to expand">
          <img className="block" src="/images/framer/pop-slide/1.png" alt="Hero Light" />
        </Accordion>
    3. Popover or Slideover:

        </Accordion>

        <Accordion title="Slideover">
          Add the following code in the `Start of <body> tag` section.

    4. Select the button or element where you want to trigger the form. Open the `properties` of the element and scroll down to `Code Overrides` section.
        1. Click on `New File` under `File` selection menu.
            <Accordion title="Click to expand">
              <img className="block" src="/images/framer/pop-slide/2.png" alt="Hero Light" />
            </Accordion>
        2. Rename the file and click on `Create`.
        3. Add the following code in the editor.
            ```TSX
            export function withClass(Component): ComponentType {
                return (props) => {
                    props.className += " surface-form-button" // Remember to add a space
                    return <Component {...props} />
                }
            }
            ```
        4. Now go back to the element, In `File` selection menu, select the file you just created and select `withClass` from the `Override` selection menu.
            <Accordion title="Click to expand">
              <img className="block" src="/images/framer/pop-slide/3.png" alt="Hero Light" />
            </Accordion>

  <Tab title="Trigger from Input">
    1. First, copy the Framer component below which includes an input field and submit button. You can customize its appearance in Framer. Paste the component anywhere in your framer project.
        - <FramerScript title={"Copy Input Component"} />

    2. Add the following code in the `End of <body> tag` section:
        ```javascript
        <script>
            const surface_src = "form_url_here"; // Replace form_url_here with your form URL

            const surface_embed_type = "popup";
            const surface_target_element_class = "surface-form-button";

            const cSurfaceForm = new SurfaceEmbed(surface_src, surface_embed_type, surface_target_element_class);

            cSurfaceForm.popupSize = "medium"; // Can be "small", "medium", or "large"
            cSurfaceForm.embedSurfaceForm();

            // Function to attach event handlers
            function attachFormHandlers() {
                const forms = document.querySelectorAll('form.surface-form-handler');
                forms.forEach((form) => {
                    if (!form.dataset.listenerAttached) {
                        form.dataset.listenerAttached = 'true';

                        form.addEventListener('submit', function (event) {
                            event.preventDefault();

                            const emailInput = form.querySelector('input[type="email"]');
                            const emailValue = emailInput?.value.trim();

                            const emailRegex = /^[^\s@]+@[^\s@]+\.[a-zA-Z]{2,}$/;

                            if (emailInput && emailValue && emailRegex.test(emailValue)) {
                                const params = {
                                    zYTngVeSzJFX_emailAddress: emailValue,
                                };

                                cSurfaceForm.showSurfaceForm(params);
                            } else {
                                console.log("Invalid email address.");
                                emailInput?.reportValidity();
                            }
                        });

                        form.addEventListener('keydown', function (event) {
                            if (event.key === 'Enter') {
                                const activeElement = document.activeElement;
                                if (activeElement.tagName === 'INPUT' && activeElement.type === 'email') {
                                    event.preventDefault();
                                    form.dispatchEvent(new Event('submit', { cancelable: true }));
                                }
                            }
                        });
                    }
                });
            }
            attachFormHandlers();

            const observer = new MutationObserver(attachFormHandlers);
            observer.observe(document.body, { childList: true, subtree: true });

            document.body.addEventListener('keydown', function (event) {
                const form = event.target.closest('form.surface-form-handler');
                if (!form) return;

                if (event.key === 'Enter') {
                    const activeElement = document.activeElement;
                    if (activeElement.tagName === 'INPUT') {
                        event.preventDefault();
                        form.dispatchEvent(new Event('submit', { cancelable: true })); 
                    }
                }
            });
        </script>
        ```

    That's it! The input component is now connected to your Surface Form.
  </Tab>
</Tabs>
